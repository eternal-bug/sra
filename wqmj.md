## 20150216: template

```bash
WORKING_DIR=${HOME}/data/dna-seq/wqmj
BASE_NAME=20150216

cd ${WORKING_DIR}/${BASE_NAME}

anchr template \
    . \
    --basename ${BASE_NAME} \
    --queue mpi \
    --genome 4000000 \
    --is_euk \
    --trim2 "--dedupe --tile" \
    --cov2 "40 80 all" \
    --qual2 "20 25 30" \
    --len2 "60" \
    --filter "adapter,phix,artifact" \
    --tadpole \
    --mergereads \
    --ecphase "1,2,3" \
    --insertsize \
    --sgapreqc \
    --sgastat \
    --fillanchor \
    --parallel 24

```
