## Switzerland: template


```bash
WORKING_DIR=${HOME}/data/dna-seq/aphid
BASE_NAME=Switzerland

cd ${WORKING_DIR}/${BASE_NAME}

rm -f *.sh
anchr template \
    . \
    --basename ${BASE_NAME} \
    --queue mpi \
    --genome 541_692_000 \
    --fastqc \
    --kmergenie \
    --insertsize \
    --sgapreqc \
    --trim2 "--dedupe" \
    --qual2 "25" \
    --len2 "60" \
    --filter "adapter,phix,artifact" \
    --mergereads \
    --ecphase "1,2,3" \
    --cov2 "all" \
    --tadpole \
    --statp 1 \
    --redoanchors \
    --fillanchor \
    --xmx 110g \
    --parallel 24

```
