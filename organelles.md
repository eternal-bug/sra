## m07: template

```bash
WORKING_DIR=${HOME}/data/dna-seq/xjy
BASE_NAME=m07

cd ${WORKING_DIR}/${BASE_NAME}

anchr template \
    . \
    --basename ${BASE_NAME} \
    --queue mpi \
    --genome 5000000 \
    --is_euk \
    --trim2 "--dedupe --tile" \
    --cov2 "10 20 40 80 160" \
    --qual2 "25 30" \
    --len2 "60" \
    --filter "adapter,phix,artifact" \
    --tadpole \
    --mergereads \
    --ecphase "1,2,3" \
    --insertsize \
    --fillanchor \
    --parallel 24

```

```bash
WORKING_DIR=${HOME}/data/dna-seq/xjy
BASE_NAME=mt302

cd ${WORKING_DIR}/${BASE_NAME}

anchr template \
    . \
    --basename ${BASE_NAME} \
    --queue mpi \
    --genome 5000000 \
    --is_euk \
    --trim2 "--dedupe --tile" \
    --cov2 "10 20 40 80 160" \

    --filter "adapter,phix,artifact" \
    --tadpole \
    --mergereads \
    --ecphase "1,2,3" \
    --insertsize \
    --fillanchor \
    --parallel 24
```
